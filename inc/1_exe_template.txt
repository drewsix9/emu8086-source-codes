.model small
.data
    pkey db "press any key...$"

.code
main proc near
    mov ax, data
    mov ds, ax
    mov es, ax

    
            
    lea dx, pkey
    mov ah, 9
    int 21h        ; output string at ds:dx
      
    mov ah, 1
    int 21h
    
    mov ax, 4c00h ; exit to operating system.
    int 21h    

main endp
end main ; set entry point and stop the assembler.
